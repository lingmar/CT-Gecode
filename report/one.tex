\PROCEDURE IndexOfFixed(): int
\STATE $\IndexOfFixed = -1$
\IF{$\Limit = 0$}
  \STATE $\Offset \leftarrow \Index[0]$
  \STATE $w \leftarrow \Words[\Offset]$
  \IF[Exactly one bit set]{$(w \ \& \ (w - 1)) = 0$}  
    \STATE $\IndexOfFixed = \Offset \cdot 64 + \textsc{MSB}(\Words[\Offset])$
  \ENDIF
\ENDIF
\RETURN $\IndexOfFixed$
